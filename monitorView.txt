<script>
    // Update the last updated time every minute
    function updateTime() {
        const now = new Date();
        document.getElementById('lastUpdated').textContent = 
            now.toLocaleDateString('en-US', { 
                month: 'long', 
                day: 'numeric', 
                year: 'numeric' 
            }) + ' ' + 
            now.toLocaleTimeString('en-US', { 
                hour: '2-digit', 
                minute: '2-digit' 
            });
    }
    
    setInterval(updateTime, 60000);
    updateTime();
</script>
